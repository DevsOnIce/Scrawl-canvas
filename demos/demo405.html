Scrawl.js - demonstration 405

Scrawl image filters - threshold()

Modules: images, filters, animation

Threshold value: 0 ('0%') 1 ('100%')

var mycode = function(){
	//define variables
	var myInput,
		updateInput;
		
	//import image
	scrawl.getImagesByClass('demo405');
		
	//clone image and add filter to it
	scrawl.image.parrot.clone({
		name: 'thresholdparrot',
		}).filter('threshold', {
			value: 0.5,
			useSourceData: true,
			});

	//define sprites
	scrawl.newPicture({
		name: 'parrot',
		startX: 10,
		startY: 10,
		scale: 0.5,
		source: 'parrot',
		}).clone({
		startX: 120,
		startY: 210,
		source: 'thresholdparrot',
		});
		
	//preparing the DOM input elements
	myInput = document.getElementById('myvalue');
	myInput.value = 0.5;

	//event listener
	updateInput = function(e){
		scrawl.image.thresholdparrot.filter('threshold', {
			value: parseFloat(myInput.value),
			useSourceData: true,
			});
		e.preventDefault();
		e.returnValue = false;
		};
	myInput.addEventListener('input', updateInput, false);	//for firefox real-time updating
	myInput.addEventListener('change', updateInput, false);
	
	//animation object
	scrawl.newAnimation({
		fn: function(){
			scrawl.render();
			},
		});
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['images', 'filters', 'animation'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
