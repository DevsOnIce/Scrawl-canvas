Scrawl.js - demonstration 003

Importing a canvas into the Scrawl library

Modules: phrase

It looks like your browser doesn't support the <canvas> element.

 

<body>
	<canvas id="mycanvas" width="400" height="200">
		<p>It looks like your browser doesn't support the <canvas> element.</p>
	</canvas>
	<script src="js/scrawlCore.js"></script>
	<script>
		var mycode = function(){
			//define some phrase sprites to hold the text
			scrawl.newPhrase({
				text: 'Hello, Scrawl World!',
				handleX: 'center',
				handleY: 'center',
				startX: 200,
				startY: 70,
				font: '30pt serif',
				method: 'fillDraw',
				fillStyle: 'red',
				}).clone({
				text: 'canvas id: '+scrawl.canvas.mycanvas.id+
						'\nwidth: '+scrawl.canvas.mycanvas.getAttribute('width')+
						'px\nheight: '+scrawl.canvas.mycanvas.getAttribute('height')+'px',
				textAlign: 'center',
				method: 'fill',
				fillStyle: 'darkblue',
				font: '12pt monospace',
				startY: 130,
				});
				
			//update the canvas display
			scrawl.render();
			};
			
		scrawl.loadModules({
			path: 'js/',
			modules: 'phrase',
			callback: function(){
				window.onload = function(){
					scrawl.init();
					mycode();
					};
				},
			});
	</script>
</body>
