Scrawl.js - demonstration 020

Scrawl Path/Shape sprites - straight lines

Modules: path, factories

var mycode = function(){

	//makeLine factory
	scrawl.makeLine({
		startX:	50,
		startY:	50,
		endX:	350,
		endY:	50,
		});

	//makeRegularShape factory, using sides
	scrawl.makeRegularShape({
		startX:	200,
		startY:	100,
		radius:	150,
		sides:	2,
		});
		
	//makeRegularShape factory, using angles
	scrawl.makeRegularShape({
		startX:	200,
		startY:	150,
		radius:	150,
		angle:	180,
		roll:	10,
		});
		
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['path', 'factories'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
