Scrawl.js - demonstration 023

Scrawl Path/Shape sprites - triangles

Modules: path, factories

var mycode = function(){
		
	//makeRegularShape factory - using sides
	scrawl.makeRegularShape({
		startX:	100,
		startY:	100,
		radius:	90,
		sides:	3,
		});
		
	//makeRegularShape factory - using angles
	scrawl.makeRegularShape({
		startX:	300,
		startY:	100,
		radius:	90,
		angle:	120,
		});
		
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['path', 'factories'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
