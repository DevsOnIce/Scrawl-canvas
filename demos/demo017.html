Scrawl.js - demonstration 017

Using the Picture sprite 'copy' attributes

Modules: images

<body>
	<img id="peacock" src="img/carousel/peacock.png" class="demo017" />
	<canvas id="mycanvas" width="400" height="200"></canvas>
	<script src="js/scrawlCore.js"></script>
	<script>
		var mycode = function(){
			//import images into scrawl library
			scrawl.getImagesByClass('demo017');

			//define the sprite
			scrawl.newPicture({
				source:		'peacock',
				width:		400,
				height:		200,
				copyX:		90,
				copyY:		130,
				copyWidth:	400,
				copyHeight:	200,
				});
				
			//display the canvas
			scrawl.render();
			};

		scrawl.loadModules({
			path: 'js/',
			modules: 'images',
			callback: function(){
				window.onload = function(){
					scrawl.init();
					mycode();
					};
				},
			});
	</script>
</body>
