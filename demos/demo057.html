Scrawl.js - demonstration 057

Scrawl animation - building Shape sprites dynamically

Modules: animation, shape, color

Click-and-drag to draw lines

var mycode = function(){
	//define variables
	var mouseDown = false, 
		currentSprite = false,
		myPad = scrawl.pad.mycanvas, 
		myCanvas = scrawl.canvas.mycanvas,
		here,
		sX, 
		sY,
		myColor,
		lines,
		startDrawing,
		keepDrawing,
		endDrawing,
		spinLines;
	
	//setup canvas
	scrawl.cell[myPad.base].set({backgroundColor: '#eee',});
	
	//define designs (color)
	myColor = scrawl.newColor({
		rMin:	50,
		rMax:	220,
		gMin:	50,
		gMax:	220,
		bMin:	50,
		bMax:	220,
		a:		1,
		});
	
	//define groups
	lines = scrawl.newGroup({
		name: 'lines',
		});
		
	//event listeners
	startDrawing = function(e){
		mouseDown = true;
		currentSprite = scrawl.newShape({
			start:			here.getData(),
			method:			'draw',
			lineWidth:		10,
			lineJoin:		'round',
			lineCap:		'round',
			strokeStyle:	myColor.clone({random: true, a: 1}).name,
			data:			'l0,0 ',
			group:			'lines',
			});
		sX = here.x;
		sY = here.y;
		if(e){
			e.stopPropagation();
			e.preventDefault();
			}
		};
	scrawl.canvas.mycanvas.addEventListener('mousedown', startDrawing, false);
		
	endDrawing = function(e){
		if(currentSprite){
			currentSprite = false;
			}
		mouseDown = false;
		if(e){
			e.stopPropagation();
			e.preventDefault();
			}
		};
	scrawl.canvas.mycanvas.addEventListener('mouseup', endDrawing, false);
	
	//animation functions
	keepDrawing = function(){
		if(currentSprite){
			if(here.x !== sX || here.y !== sY){
				currentSprite.set({
					data: currentSprite.data+' '+(here.x - sX)+','+(here.y - sY),
					});
				sX = here.x;
				sY = here.y;
				}
			}
		};
	spinLines = function(){
		for(var i = 0, z = lines.sprites.length; i < z; i++){
			scrawl.sprite[lines.sprites[i]].setDelta({
				roll: 1,
				});
			}
		};
	
	//animation object
	scrawl.newAnimation({
		fn: function(){
			here = myPad.getMouse();
			if(here.active){
				(mouseDown) ? keepDrawing() : spinLines();
				}
			else{
				if(currentSprite){
					endDrawing();
					}
				spinLines();
				}
			scrawl.render();
			},
		});
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['animation', 'shape', 'color'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
