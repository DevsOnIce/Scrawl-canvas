Scrawl.js - demonstration 030

Rounded corner rectangle Shape sprites

Modules: path, factories

var mycode = function(){
	//makeRectangle factory
	scrawl.makeRectangle({
		startX:			200,
		startY:			80,
		width:			200,
		height:			100,
		radius:			20,
		lineWidth:		5,
		fillStyle:		'blue',
		strokeStyle:	'red',
		method:			'fillDraw',
		});
	scrawl.makeRectangle({
		startX:			200,
		startY:			200,
		width:			350,
		height:			60,
		radiusTop:		20,
		lineWidth:		5,
		fillStyle:		'blue',
		strokeStyle:	'red',
		method:			'fillDraw',
		});
	scrawl.makeRectangle({
		startX:			200,
		startY:			320,
		width:			100,
		height:			100,
		radius:			100,
		lineWidth:		5,
		lineJoin:		'round',
		fillStyle:		'blue',
		strokeStyle:	'red',
		method:			'fillDraw',
		});
		
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['path', 'factories'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
