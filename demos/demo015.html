		<canvas id="mycanvas" width="400" height="200"></canvas>
		<script src="js/scrawlCore.js"></script>
		<script>
			var mycode = function(){
				//define sprite ...
				scrawl.newPhrase({
					startX:		200,
					startY:		100,
					handleX:	'center',
					handleY:	'center',
					size:		80,
					metrics:	'px',
					family:		'"KaiTi", "楷体", STKaiti, "华文楷体", "Microsoft YaHei New", "Microsoft Yahei", "微软雅黑", STXihei, "华文细黑", SimSun, 宋体, Georgia, "Times New Roman", serif',
					text:		'你好',
					});
					
				//... and display it
				scrawl.render();
				};
				
			scrawl.loadModules({
				path: 'js/',
				modules: 'phrase',
				callback: function(){
					window.onload = function(){
						scrawl.init();
						mycode();
						};
					},
				});
		</script>
    </body>
