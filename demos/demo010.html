Scrawl.js - demonstration 010

Styling the Wheel sprite, and its clone

Modules: wheel

var mycode = function(){
	//define sprites
	scrawl.newWheel({
		name: 'mywheel',
		startX:			100,
		startY:			100,
		radius:			80,
		fillStyle:		'Gold',
		strokeStyle:	'rgb(0,190,0)',
		lineWidth:		10,
		lineCap:		'round',
		lineJoin:		'round',
		shadowOffsetX:	3,
		shadowOffsetY:	3,
		shadowBlur:		2,
		shadowColor:	'Black',
		method:			'sinkInto',
		}).clone({
		startX:			300,
		startAngle:		30,
		endAngle:		325,
		includeCenter:	true,
		scale:			0.7,
		});
		
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: 'wheel',
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
