Scrawl.js - demonstration 005

Cloning a Scrawl Block sprite

Modules: block

var mycode = function(){
	//define block sprites
	scrawl.newBlock({
		startX:		10,
		startY:		10,
		width:		180,
		height:		180,
		}).clone({
		startX:		210,
		scale:		0.8,
		lineWidth:	5,
		method:		'draw',
		});
	
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: 'block',
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
