Scrawl.js - demonstration 016

Ellipse Shape sprite

Modules: shape, factories, animation

var mycode = function(){
	//define sprite
	scrawl.makeEllipse({
		name:			'ellie',
		startX:			200,
		startY:			200,
		radiusX:		150,
		radiusY:		50,
		lineWidth:		5,
		fillStyle:		'blue',
		strokeStyle:	'red',
		method:			'fillDraw',
		shape:			true,
		});
		
	//animation object
	scrawl.newAnimation({
		fn: function(){
			scrawl.sprite.ellie.setDelta({
				roll: 1,
				});
			scrawl.cell.mycanvas_base.set({
				backgroundColor: (scrawl.sprite.ellie.checkHit(scrawl.pad.mycanvas.getMouse())) ? 'lightblue' : 'lightgreen',
				});
			scrawl.render();
			},
		});
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['shape', 'factories', 'animation'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
