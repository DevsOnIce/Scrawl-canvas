Scrawl.js - demonstration 001

Adding a Scrawl.js canvas to a page programmatically - #1

Modules: phrase

Canvas will appear at bottom of this page ...

<!DOCTYPE html>
<html>
    <head>
        <title>Scrawl.js</title>
	</head>
    <body>
		<p>Canvas will appear at bottom of this page ...</p>
		<script src="js/scrawlCore.js"></script>
		<script>
			var mycode = function(){
				//add a canvas to the end of document.body, and make it current
				scrawl.addCanvasToPage({
					width:	400,
					height:	200,
					}).makeCurrent();
					
				//define a phrase sprite to hold some text
				scrawl.newPhrase({
					text:	'Hello, Scrawl World!',
					startX:	50,
					startY:	50,
					});
					
				//update the canvas display
				scrawl.render();
				};
				
			scrawl.loadModules({
				path:		'js/',				//path to where Scrawl modules are stored, relative to this web page
				modules:	'phrase',			//array (or string) of Scrawl modules to be loaded 
				callback:	function(){			//function to perform once all the Scrawl modules have loaded
					window.onload = function(){
						scrawl.init();			//start Scrawl
						mycode();				//run the mycode function defined above
						};
					},
				});
		</script>
    </body>
</html>
