Scrawl.js - demonstration 024

Scrawl Path/Shape sprites - pentangle (or star)

Modules: path, factories

var mycode = function(){
		
	//makeRegularShape factory
	scrawl.makeRegularShape({
		startX:	200,
		startY:	100,
		radius:	90,
		angle:	144,
		});
		
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['path', 'factories'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
