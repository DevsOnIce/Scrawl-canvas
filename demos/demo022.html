Scrawl.js - demonstration 022

Scrawl Path/Shape sprites - bezier curve

Modules: path, factories

var mycode = function(){
		
	//makeBezier factory
	scrawl.makeBezier({
		startX:			50,
		startY:			100,
		endX:			350,
		endY:			100,
		startControlX:	150,
		startControlY:	0,
		endControlX:	250,
		endControlY:	200,
		});
		
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['path', 'factories'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
