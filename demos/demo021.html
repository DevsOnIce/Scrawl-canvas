Scrawl.js - demonstration 021

Scrawl Path/Shape sprites - quadratic curve

Modules: path, factories

var mycode = function(){
		
	//makeQuadratic factory
	scrawl.makeQuadratic({
		startX:		50,
		startY:		150,
		endX:		350,
		endY:		150,
		controlX:	200,
		controlY:	0,
		});
		
	//display canvas
	scrawl.render();
	};
	
scrawl.loadModules({
	path: 'js/',
	modules: ['path', 'factories'],
	callback: function(){
		window.onload = function(){
			scrawl.init();
			mycode();
			};
		},
	});
