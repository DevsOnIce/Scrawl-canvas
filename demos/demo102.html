Scrawl.js - demonstration 102

Apply a filter to an animated Picture sprite

Modules: images, animation, filters

<body>
	<img id="runningcat" src="img/runningcat.png" />
	<canvas id="mycanvas" width="400" height="200"></canvas>
	<script src="js/scrawlCore.js"></script>
	<script>
		var mycode = function(){
			//load spritesheet into scrawl library
			scrawl.getImageById('runningcat');
		
			//define AnimSheet object
			scrawl.newAnimSheet({
				name:		'animatedCat',
				running:	'forward',
				loop:		'loop',
				speed:		1.2,
				frames:		[
					{x: 0,		y: 0,		w: 512,		h: 256,		d: 100,},
					{x: 512,	y: 0,		w: 512,		h: 256,		d: 100,},
					{x: 0,		y: 256,		w: 512,		h: 256,		d: 100,},
					{x: 512,	y: 256,		w: 512,		h: 256,		d: 100,},
					{x: 0,		y: 512,		w: 512,		h: 256,		d: 100,},
					{x: 512,	y: 512,		w: 512,		h: 256,		d: 100,},
					{x: 0,		y: 768,		w: 512,		h: 256,		d: 100,},
					{x: 512,	y: 768,		w: 512,		h: 256,		d: 100,},
					],
				});
				
			//define Picture sprite which will display animation
			scrawl.newPicture({
				width:		400,
				height:		200,
				method:		'fill',
				source:		'runningcat',
				animSheet:	'animatedCat',
				filters:	{
					glassTile: {
						width: 20,
						height: 20,
						outerWidth: 30,
						outerHeight: 30,
						},
					},
				});
			
			//animation object
			scrawl.newAnimation({
				fn: function(){
					scrawl.render();
					},
				});
			};

		scrawl.loadModules({
			path: 'js/',
			modules: ['images', 'animation', 'filters'],
			callback: function(){
				window.onload = function(){
					scrawl.init();
					mycode();
					};
				},
			});
	</script>
</body>
